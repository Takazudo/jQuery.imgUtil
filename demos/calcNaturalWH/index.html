$.imgUtil.calcNaturalWH

$.imgUtil.calcNaturalWH caliculates thrown img's natural width and height gracefully.

$.imgUtil.calcNaturalWH('../imgs/1.jpg').done(function(wh, $img){
  $('#test1-info').append('<li>1.jpg width:' + wh.width + '</li>');
  $('#test1-info').append('<li>1.jpg height:' + wh.height + '</li>');
  $('#test1-imgs').append($img);
});

});
