$.fn.imgCoverRect

$.fn.imgCoverRect oninit option.

$('.imgrect').imgCoverRect({
  oninit: function(data) {
    // spin.js http://fgnass.github.io/spin.js/
    var spinnerOptions = {
      color: '#333',
      lines: 10,
      length: 10,
      width: 2,
      radius: 10
    };
    (new Spinner(spinnerOptions)).spin(data.el[0]);
  }
});
